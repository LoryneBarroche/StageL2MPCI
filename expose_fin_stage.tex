\begin{frame}
\frametitle{Qu'est-ce que Rocq ?}

\begin{itemize}
    \item Un assistant de preuves, c-à-d un langage de programmation permettant la formalisation de preuves mathématiques \\
    \item Formaliser une preuve = écrire une preuve de telle sorte à ce que chaque argument découle sans ambiguïté des faits précédents. \\
    Formaliser $\implies$ comprendre tous les détails implicites !
\end{itemize}

\end{frame}


\begin{frame}
\frametitle{Pavage du plan avec des tuiles de Wang}

\begin{itemize}
    \item \textbf{Pavage du plan :}  un ensemble de portions du plan recouvrant le plan tout entier sans interstices ni chevauchements\\
    \item \textbf{Tuiles de Wang :}  des carrés de longueur 1 définies par la couleur de chacun de ses côtés. Un ensemble fini de tuiles de Wang sera noté $\tau$.\\
    \item \textbf{Configuration :}  une fonction $P : \Z^2 \to \tau$ \\
    \item \textbf{Pavage du plan avec des tuiles de Wang :}   une configuration $P$ qui est \guillemotleft valide \guillemotright (c-à-d recouvrement du plan et compatibilité).
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Périodicité sur les pavages de Wang}
    \begin{itemize}
        \item  $P : \Z^2 \rightarrow \tau$ est \textbf{faiblement périodique} ssi $$\exists \Vec{u}  \in \Z^2 \setminus \{\begin{pmatrix} 0 \\ 0 \end{pmatrix}\} , \forall x \in \Z^2, \forall k \in \Z, P(x+k\Vec{u}) = P(x)$$\\
        $\Vec{u}$ est un vecteur de périodicité de $P$.
        \item $P : \Z^2 \rightarrow \tau$ est \textbf{fortement périodique} si et seulement si il existe $\Vec{u}$ et $\Vec{v}$ deux vecteurs de périodicité de $P$ non colinéaires.
    \end{itemize}

\pause

A-t-on fortement périodique $\Leftrightarrow$ faiblement périodique ?\\
Le sens $\Rightarrow$ est évident par définition, mais le sens $\Leftarrow$ l'est beaucoup moins.
\end{frame}

\begin{frame}{Premiers résultats en Rocq}

\begin{thm}{}{}
Si $P$ est une configuration admettant $\begin{pmatrix} 1 \\ 1 \end{pmatrix}$ comme vecteur de périodicité
et telle que tout vecteur colinéaire à $\begin{pmatrix} 1 \\ 0 \end{pmatrix}$ n'est pas un vecteur de
périodicité, alors P n'est pas fortement périodique.
\end{thm}
\end{frame}

\begin{frame}{Premiers résultats en Rocq}
    


\begin{thm}{}{}
Si $P$ est fortement périodique, alors il admet un vecteur de périodicité colinéaire à $\begin{pmatrix} 1 \\ 0 \end{pmatrix}$.
\end{thm}

\end{frame}


\begin{frame}

    \frametitle{Théorème}
    
\begin{thm}{}{}
Soit $\tau$ un jeu fini de tuiles de Wang en deux dimensions.\\
$\exists P : \Z^2 \rightarrow \tau$ valide, $P$ faiblement périodique ssi $\exists P_D : \Z^2 \rightarrow \tau$ valide, $P_D$ fortement périodique. \\ 
\end{thm}
\end{frame}
